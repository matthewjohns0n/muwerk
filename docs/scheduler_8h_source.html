274  bool mqttmatch(const String pubstr, const String substr) {
283  if (pubstr == substr)
284  return true;
285 
286  // Attiny compile: core needs to be extended, add c_str():
287  // In WString.h add: const char *c_str() const { return buffer; }
288  const char *pub = (const char *)pubstr.c_str();
289  const char *sub = (const char *)substr.c_str();
290 
291  int lp = strlen(pub);
292  int ls = strlen(sub);
293 
294  bool wPos = true; // sub wildcard is legal now
295  int ps = 0;
296  for (int pp = 0; pp < lp; pp++) {
297  if (pub[pp] == '+' || pub[pp] == '#') {
298  return false; // Illegal wildcards in pub
299  }
300  if (wPos) {
301  wPos = false;
302  if (sub[ps] == '#') {
303  if (ps == ls - 1) {
304  return true;
305  } else {
306  return false; // In sub, # must not be followed by
307  // anything else
308  }
309  }
310  if (sub[ps] == '+') {
311  while (pp < lp && pub[pp] != '/')
312  ++pp;
313  ++ps;
314  if (pp == lp) {
315  if (ps == ls) {
316  return true;
317  } else if (!strcmp(&sub[ps], "/#")) {
318  return true;
319  }
320  }
321  }
322  } else {
323  if (sub[ps] == '+' || sub[ps] == '#') {
324  return false; // Illegal wildcard-position
325  }
326  }
327  if (pub[pp] != sub[ps] && strcmp(&sub[ps], "/#")) {
328  return false;
329  }
330  if (pub[pp] == '/')
331  wPos = true;
332  if (pp == lp - 1) {
333  if (ps == ls - 1) {
334  return true;
335  }
336  if (!strcmp(&sub[ps + 1], "/#") || !strcmp(&sub[ps + 1], "#") ||
337  !strcmp(&sub[ps + 1], "+")) {
338  return true;
339  }
340  return false;
341  }
342  ++ps;
343  }
344  if (ps == ls) {
345  return true;
346  } else {
347  return false;
348  }
