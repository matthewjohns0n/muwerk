314  static bool mqttmatch(const String pubstr, const String substr) {
323  if (pubstr == substr)
324  return true;
325 
326  // Attiny compile: core needs to be extended, add c_str():
327  // In WString.h add: const char *c_str() const { return buffer; }
328  const char *pub = (const char *)pubstr.c_str();
329  const char *sub = (const char *)substr.c_str();
330 
331  int lp = strlen(pub);
332  int ls = strlen(sub);
333 
334  bool wPos = true; // sub wildcard is legal now
335  int ps = 0;
336  for (int pp = 0; pp < lp; pp++) {
337  if (pub[pp] == '+' || pub[pp] == '#') {
338  return false; // Illegal wildcards in pub
339  }
340  if (wPos) {
341  wPos = false;
342  if (sub[ps] == '#') {
343  if (ps == ls - 1) {
344  return true;
345  } else {
346  return false; // In sub, # must not be followed by
347  // anything else
348  }
349  }
350  if (sub[ps] == '+') {
351  while (pp < lp && pub[pp] != '/')
352  ++pp;
353  ++ps;
354  if (pp == lp) {
355  if (ps == ls) {
356  return true;
357  } else if (!strcmp(&sub[ps], "/#")) {
358  return true;
359  }
360  }
361  }
362  } else {
363  if (sub[ps] == '+' || sub[ps] == '#') {
364  return false; // Illegal wildcard-position
365  }
366  }
367  if (pub[pp] != sub[ps] && strcmp(&sub[ps], "/#")) {
368  return false;
369  }
370  if (pub[pp] == '/')
371  wPos = true;
372  if (pp == lp - 1) {
373  if (ps == ls - 1) {
374  return true;
375  }
376  if (!strcmp(&sub[ps + 1], "/#") || !strcmp(&sub[ps + 1], "#") ||
377  !strcmp(&sub[ps + 1], "+")) {
378  return true;
379  }
380  return false;
381  }
382  ++ps;
383  }
384  if (ps == ls) {
385  return true;
386  } else {
387  return false;
388  }
389  }
