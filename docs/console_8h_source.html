216  void loop() {
217  int incomingByte;
218  bool changed = false;
219  int count = 0;
220 
221  while ((incomingByte = Serial.read()) != -1 && count < 16) {
222  ++count; // limit reads per cycle
223  switch (incomingByte) {
224  case 0: // ignore
225  case 10: // ignore
226  break;
227  case 8: // backspace
228  if (pcur > buffer) {
229  --pcur;
230  *pcur = 0;
231  } else if (args.length()) {
232  args.remove(args.length() - 1, 1);
233  }
234  changed = true;
235  break;
236  case 13: // enter
237  flush();
238  execute();
239  changed = true;
240  break;
241  case 9: // tab
242  incomingByte = 32;
243  // treat like space
244  default:
245  *pcur = (char)incomingByte;
246  ++pcur;
247  if (pcur == buffer + (sizeof(buffer) / sizeof(char)) - 1) {
248  // buffer full - flush it to args
249  flush();
250  }
251  changed = true;
252  break;
253  }
254  }
255 
256  if (changed) {
257  prompt();
258  }
259  }
