189  while ((incomingByte = Serial.read()) != -1 && count < 16) {
190  ++count; // limit reads per cycle

193  case 10: // ignore
194  break;
195  case 8: // backspace

197  --pcur;
198  *pcur = 0;

200  changed = true;
201  break;
202  case 13: // enter
203  flush();
204  execute();
205  changed = true;
206  break;
207  case 9: // tab
208  incomingByte = 32;
209  // treat like space
210  default:
211  *pcur = (char)incomingByte;

214  // buffer full - flush it to args
