189  void loop() {
190  int incomingByte;
191  bool changed = false;
192  int count = 0;
193 
194  while ((incomingByte = Serial.read()) != -1 && count < 16) {
195  ++count; // limit reads per cycle
196  switch (incomingByte) {
197  case 0: // ignore
198  case 10: // ignore
199  break;
200  case 8: // backspace
201  if (pcur > buffer) {
202  --pcur;
203  *pcur = 0;
204  }
205  changed = true;
206  break;
207  case 13: // enter
208  flush();
209  execute();
210  changed = true;
211  break;
212  case 9: // tab
213  incomingByte = 32;
214  // treat like space
215  default:
216  *pcur = (char)incomingByte;
217  ++pcur;
218  if (pcur == buffer + (sizeof(buffer) / sizeof(char)) - 1) {
219  // buffer full - flush it to args
220  flush();
221  }
222  changed = true;
223  break;
224  }
225  }

229  }
