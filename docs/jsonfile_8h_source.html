923  filename = basename;
924  fs::File f = fsOpen(fn, "r");
925  if (!f) {

934  // Lets read line by line from the file

938  f.close();

940  if (JSON.typeof(content) == "undefined") {
941  DBG("Parsing input file " + fn + "failed, invalid JSON!");
942  DBG2("Content: " + jsonstr);
943  return false;
944  }
945  DBG2("Input file " + fn + " successfully parsed");
946  DBG3("Content: " + jsonstr);
947  obj = content;
948  loaded = true;
949  return true;

963  bool prepareRead(String key, ustd::array<String> &keyparts, JSONVar &subobj,
964  bool objmode = false) {
965  normalize(key);
966  ustd::split(key, '/', keyparts);
967  if (keyparts.length() < (objmode ? 1 : 2)) {
968  DBG("Key-path too short, minimum needed is filename/topic, got: " + key);
969  return false;
970  }
971  if (!checkLoad(keyparts[0])) {
972  return false;
973  }
974  JSONVar iterator(obj);
975  for (unsigned int i = 1; i < keyparts.length() - 1; i++) {
976  JSONVar tmpCopy(iterator[keyparts[i]]);
977  iterator = tmpCopy;
978  if (JSON.typeof(iterator) == "undefined") {
979  DBG2("From " + key + ", element " + keyparts[i] + " not found.");
980  return false;
981  }
982  }
983  String lastKey = keyparts[keyparts.length() - 1];
984  if (!iterator.hasOwnProperty(lastKey)) {
985  DBG2("From " + key + ", element " + lastKey + " not found.");
986  return false;
987  } else {
988  JSONVar tmpCopy(iterator[lastKey]);
989  subobj = tmpCopy;
990  }
991  return true;
992  }

994  bool prepareWrite(String key, JSONVar &target, bool objmode = false) {
995  ustd::array<String> keyparts;
996 
997  normalize(key);
998  ustd::split(key, '/', keyparts);
999  if (keyparts.length() < (objmode ? 1 : 2)) {
1000  DBG("Key-path too short, minimum needed is filename/topic, got: " + key);
1001  return false;
1002  }
1003  if (keyparts.length() > MAX_FRICKEL_DEPTH) {
1004  DBG("Key-path too long, maxdepth is " + String(MAX_FRICKEL_DEPTH) + ", got: " + key);
1005  return false;
1006  }
1007  if (!checkLoad(keyparts[0]) && forcenew) {
1008  DBG("Creating new file /" + keyparts[0] + ".json");
1009  }
1010 
1011  // frickel
1012  switch (keyparts.length()) {
1013  case 1:
1014  // possible only in object mode
1015  target = obj;
1016  return true;
1017  case 2:
1018  target = obj[keyparts[1]];
1019  return true;
1020  case 3:
1021  target = obj[keyparts[1]][keyparts[2]];
1022  return true;
1023  case 4:
1024  target = obj[keyparts[1]][keyparts[2]][keyparts[3]];
1025  return true;
1026  case 5:
1027  target = obj[keyparts[1]][keyparts[2]][keyparts[3]][keyparts[4]];
1028  return true;
1029  case 6:
1030  target = obj[keyparts[1]][keyparts[2]][keyparts[3]][keyparts[4]][keyparts[5]];
1031  return true;
1032  case 7:
1033  target =
1034  obj[keyparts[1]][keyparts[2]][keyparts[3]][keyparts[4]][keyparts[5]][keyparts[6]];
1035  return true;
1036  case 8:
1037  target = obj[keyparts[1]][keyparts[2]][keyparts[3]][keyparts[4]][keyparts[5]]
1038  [keyparts[6]][keyparts[7]];
1039  return true;
1040  case 9:
1041  target = obj[keyparts[1]][keyparts[2]][keyparts[3]][keyparts[4]][keyparts[5]]
1042  [keyparts[6]][keyparts[7]][keyparts[8]];
1043  return true;
1044  default:
1045  DBG("SERIOUS PROGRAMMING ERROR - MAX_FRICKEL_DEV higher than implemented support "
1046  "in " __FILE__ " line number " +
1047  String(__LINE__));
1048  return false;
1049  }
